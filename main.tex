 \title{First-order tree-to-tree functions}
 \author{Amina and Mikolaj}
 \begin{abstract}
    We study tree-to-tree transformations that can be defined in logics such as  first-order logic or  monadic second-order logic. We show that such transformations can be decomposed into certain primitive transformations, such as tree-to-tree homomorphisms or pre-order traversal, by using combinators such as function composition. 
\end{abstract}
